\begin{align*} \Sigma F_x \amp = 0\\ G_h - 6\,T_x \amp= 0\\ G_h \amp= 6\,(T \cos 76Â°)\\ \amp= \kN{14.5} \end{align*}

\begin{align*} G_h/G \amp= \sin \ang{38}\\ G \amp= G_h/\sin \ang{38}\\ G \amp= \kN{23.6} \end{align*}
